function SunsetHills(){


    // Step 1: Get all the user data
    let num1 = parseInt(document.getElementById("shInput1").value);
    let num2 = parseInt(document.getElementById("shInput2").value);
    let num3 = parseInt(document.getElementById("shInput3").value);
    let num4 = parseInt(document.getElementById("shInput4").value);
    let num5 = parseInt(document.getElementById("shInput5").value);


    // const buildings = document.querySelectorAll(".form-control").forEach()


    // Step 2: Turn the user data into an array
    let userArray = new Array();
    userArray.push(num1);
    userArray.push(num2);
    userArray.push(num3);
    userArray.push(num4);
    userArray.push(num5);


    // Step 3: Determine which building can see the sunset
    // By default the first number will be the max number
    let maxNum = userArray[0];
    let acendingArray = new Array();
    acendingArray.push(maxNum);


    // Step4: Cycle through the array and always add the largest numbers
    // If and only if the current building is larger than the largest previous number will I add it to the  precvious building 
    for (let loop=1; loop < userArray.length; loop++){
        if( userArray[loop] > maxNum ){

            maxNum = userArray[loop];
            acendingArray.push(maxNum);

        }
    }
    

    // Step 5: Output results to the user
    // Out to id="output" innerText or innerHTML
    document.getElementById("output").innerText = `The Ascending array is [${acendingArray.join(`, `)}]`;


    // Output to id="output" innerText or innerHTML


    // 2 3 4 6 8 3
    // [2,4,6,8,3]
    // maxNum = 2
    // ascendingArray = [2]


    // userArray[1] = 4
    // 4 > maxNum --> 4 > 2
    // maxNum = 4 
    // ascendingArray = [2, 4]
    // ...


    // userArray[4] = 3
    // 3 > maxNum --> 3 > 8
    // ascendingArray = [2,4,6,8]


    // Output to id="output" innerText or innerHTML


}


function Clear(){

    document.getElementById("shInput1").value = "";
    document.getElementById("shInput2").value = "";
    document.getElementById("shInput3").value = "";
    document.getElementById("shInput4").value = "";
    document.getElementById("shInput5").value = "";
    document.getElementById("output").innerText = "";

}
